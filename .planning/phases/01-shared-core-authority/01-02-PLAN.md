nonisolated func localized(_ key: String, locale: Locale) -> String {
    let bundle = localizationBundle(for: locale)
    let localizedValue = bundle.localizedString(forKey: key, value: nil, table: nil)
    if localizedValue != key || bundle == Bundle.main { return localizedValue }
    return Bundle.main.localizedString(forKey: key, value: key, table: nil)
}

    Per D-08, D-09, D-11: Convert the entire xcstrings catalog to KMP Multiplatform Resources.

    **Step 1 — Create the conversion script:**
    Write `scripts/convert_xcstrings_to_kmp.py` that:
    1. Reads `Pindrop/Localization/Localizable.xcstrings` JSON
    2. For each string key:
       - If the key is a short identifier (e.g., "Settings", "Cancel"), use it directly as the KMP resource name (lowercased, snake_case)
       - If the key is long English text (the Apple pattern where the English string IS the key), generate a short identifier by hashing or abbreviating. Map to something readable like `error_<topic>_<index>` or derive from the first few words
       - If the key contains format specifiers, convert: `%@` → `%s`, `%lld` → `%d`, and for multi-param strings use positional: `%1$s`, `%2$s`, `%3$s`
    3. Outputs `values/strings.xml` (English base) and `values-{locale}/strings.xml` for each locale
    4. XML escaping: ampersand → `&amp;`, angle brackets → `&lt;`/`&gt;`, apostrophe → `\'`
    5. Also handles `InfoPlist.xcstrings` — generates `values/infoplist_strings.xml` if needed

    Run the script: `python3 scripts/convert_xcstrings_to_kmp.py`

    **Step 2 — Create KMP module:**

    - Apply `kotlin("multiplatform")` and `org.jetbrains.kotlin.plugin.compose` (for resources)
    - Add `id("org.jetbrains.compose")` if needed for multiplatform-resources
    - Configure `kotlin { sourceSets { commonMain { dependencies { } } } }` with multiplatform-resources plugin
    - XCFramework: `PindropSharedLocalization` with macOS ARM64 + x64 targets
    - The resources go in `src/commonMain/resources/` which multiplatform-resources packages into the framework

    Register in `shared/settings.gradle.kts`: `include(":ui-localization")`
